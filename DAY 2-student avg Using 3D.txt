#include <stdio.h>

int main()
{
    int marks[3][2][3];
    int sem, stu, sub;
    float avg, sum;

    // Input marks
    for (sem = 0; sem < 3; sem++)
    {
        printf("\nSemester %d\n", sem + 1);
        for (stu = 0; stu < 2; stu++)
        {
            printf(" Student %d\n", stu + 1);
            for (sub = 0; sub < 3; sub++)
            {
                printf("  Enter Subject %d marks: ", sub + 1);
                scanf("%d", &marks[sem][stu][sub]);
            }
        }
    }


    printf("\n--- Average Marks ---\n");
    for (sem = 0; sem < 3; sem++)
    {
        printf("\nSemester %d\n", sem + 1);
        for (stu = 0; stu < 2; stu++)
        {
            sum = 0;
            for (sub = 0; sub < 3; sub++)
            {
                sum += marks[sem][stu][sub];
            }
            avg = sum / 3;
            printf(" Student %d Average = %.2f\n", stu + 1, avg);
        }
    }

    return 0;
}
